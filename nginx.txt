server {
    listen 443 ssl http2;
    server_name chatademia.social;

    ssl_certificate /etc/letsencrypt/live/chatademia.social/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/chatademia.social/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;

    # Frontend (React)
    root /var/www/chatademia.social;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # Backend
    location /backend/ {
	proxy_pass http://127.0.0.1:8080/;
	proxy_http_version 1.1;
	# ðŸ”‘ REQUIRED for WebSockets
	proxy_set_header Upgrade $http_upgrade;
	proxy_set_header Connection "Upgrade";
	# Standard headers
	proxy_set_header Host $host;
	proxy_set_header X-Real-IP $remote_addr;
	proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
	proxy_set_header X-Forwarded-Proto https;
	# Auth / cookies
	proxy_set_header Authorization $http_authorization;
	proxy_set_header Cookie $http_cookie;
	# IMPORTANT: disable buffering for WS
	proxy_buffering off;
    }
}
